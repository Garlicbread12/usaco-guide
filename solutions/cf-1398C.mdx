## Solution - Prefix Sums + Math

```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using pi = pair<ll, ll>;
void solve(){
	int n; cin >> n;
	vector<int> dp(n+1, 0);
	for (int i=1; i<=n; i++) {
		char c; cin >> c;
		dp[i]=c-'0';
	}
	for (int i=1; i<=n; i++) {
		dp[i]+=dp[i-1];
	}
	map<int, ll> sumDist;
	for (int i=0; i<=n; i++) {
		sumDist[dp[i]-i]++;
	}
	ll goodArrays=0;
	for (pi p: sumDist) {
		ll f=p.second;
		// sum of all positive integers less than f
		goodArrays+=f*(f-1)/2;
	}
	cout << goodArrays << endl;
}

int main() {
	int t; cin >> t;
	for (int i=0; i<t; i++) {
		solve();
	}
}
```

```java
import java.io.*;
import java.util.*;

public class GoodSubarrays {
	static long solve (int arrayLength, String stringArray) {
		int[] array = new int[arrayLength + 1];
		for (int x = 1; x <= arrayLength; x++) {  
			array[x] = stringArray.charAt(x - 1) - '0';
		}
		for (int x = 1; x <= arrayLength; x++) { 
			array[x] += array[x - 1];
		}
		Map<Integer, Long> sumDist = new HashMap<>();
		for (int x = 0; x <= arrayLength; x++) {  
			if (sumDist.containsKey(array[x] - x)) {  
				sumDist.put(array[x] - x, sumDist.get(array[x] - x) + 1);
			}
			else {  
				sumDist.put(array[x] - x, (long) 1);
			}
		}
		long goodArrays = 0;
		for (int key : sumDist.keySet()) {  
			long f = sumDist.get(key);
			// sum of all positive integers less than f
			goodArrays += f * (f - 1) / 2;
		}
		return goodArrays;
	}

	public static void main (String[] args) {
		Kattio io = new Kattio();
		int t = io.nextInt();
		for (int x = 0; x < t; x++) { 
			int arrayLength = io.nextInt();
			String array = io.next();
			io.println(solve(arrayLength, array));
		}
		io.close();
	}
	//CodeSnip{Kattio}
}
```

```py
import sys

# pretty much the official editorial sol
t_cases = int(input())
for i in range(t_cases):
	len_nums = int(input())
	nums = input().strip()
	# start with 0 at the head of list
	record = {0:1}
	ans, prefix_sum= 0, 0

	for j in range(len_nums):
		prefix_sum += int(nums[j])
		val = prefix_sum - j - 1 
		if val not in record:
			record[val] = 0
		record[val] += 1

	for id in record:
	# sum of positive integers less than 'f'
		ans += (record[id] * (record[id]-1))//2 
	print(str(ans) +  '\n')
```
